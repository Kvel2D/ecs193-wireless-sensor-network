71 // Wait until no channel activity detected or timeout
72 bool RHGenericDriver::waitCAD()
73 {
74  if (!_cad_timeout)
75  return true;
76 
77  // Wait for any channel activity to finish or timeout
78  // Sophisticated DCF function...
79  // DCF : BackoffTime = random() x aSlotTime
80  // 100 - 1000 ms
81  // 10 sec timeout
82  unsigned long t = millis();
83  while (isChannelActive())
84  {
85  if (millis() - t > _cad_timeout)
86  return false;
87 #if (RH_PLATFORM == RH_PLATFORM_STM32) // stdlib on STMF103 gets confused if random is redefined
88  delay(_random(1, 10) * 100);
89 #else
90  delay(random(1, 10) * 100); // Should these values be configurable? Macros?
91 #endif
92  }
93 
94  return true;
95 }
