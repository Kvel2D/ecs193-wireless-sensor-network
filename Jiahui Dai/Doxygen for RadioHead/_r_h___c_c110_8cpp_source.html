77 bool RH_CC110::init()
78 {
79  if (!RHNRFSPIDriver::init())
80  return false;
81 
82  // Determine the interrupt number that corresponds to the interruptPin
83  int interruptNumber = digitalPinToInterrupt(_interruptPin);
84  if (interruptNumber == NOT_AN_INTERRUPT)
85  return false;
86 #ifdef RH_ATTACHINTERRUPT_TAKES_PIN_NUMBER
87  interruptNumber = _interruptPin;
88 #endif
89 
90  // Tell the low level SPI interface we will use SPI within this interrupt
91  spiUsingInterrupt(interruptNumber);
92 
93  // Reset the chip
94  // Strobe the reset
95  uint8_t val = spiCommand(RH_CC110_STROBE_30_SRES); // Reset
96  delay(100);
97  val = spiCommand(RH_CC110_STROBE_36_SIDLE); // IDLE
98  if (val != 0x0f)
99  return false; // No chip there or reset failed.
100 
101  // Add by Adrien van den Bossche <vandenbo@univ-tlse2.fr> for Teensy
102  // ARM M4 requires the below. else pin interrupt doesn't work properly.
103  // On all other platforms, its innocuous, belt and braces
104  pinMode(_interruptPin, INPUT);
105 
106  // Set up interrupt handler
107  // Since there are a limited number of interrupt glue functions isr*() available,
108  // we can only support a limited number of devices simultaneously
109  // ON some devices, notably most Arduinos, the interrupt pin passed in is actuallt the
110  // interrupt number. You have to figure out the interruptnumber-to-interruptpin mapping
111  // yourself based on knwledge of what Arduino board you are running on.
112  if (_myInterruptIndex == 0xff)
113  {
114  // First run, no interrupt allocated yet
115  if (_interruptCount <= RH_CC110_NUM_INTERRUPTS)
116  _myInterruptIndex = _interruptCount++;
117  else
118  return false; // Too many devices, not enough interrupt vectors
119  }
120  _deviceForInterrupt[_myInterruptIndex] = this;
121  if (_myInterruptIndex == 0)
122  attachInterrupt(interruptNumber, isr0, RISING);
123  else if (_myInterruptIndex == 1)
124  attachInterrupt(interruptNumber, isr1, RISING);
125  else if (_myInterruptIndex == 2)
126  attachInterrupt(interruptNumber, isr2, RISING);
127  else
128  return false; // Too many devices, not enough interrupt vectors
129 
130  spiWriteRegister(RH_CC110_REG_02_IOCFG0, RH_CC110_GDO_CFG_CRC_OK_AUTORESET); // gdo0 interrupt on CRC_OK
131  spiWriteRegister(RH_CC110_REG_06_PKTLEN, RH_CC110_MAX_PAYLOAD_LEN); // max packet length
132  spiWriteRegister(RH_CC110_REG_07_PKTCTRL1, RH_CC110_CRC_AUTOFLUSH); // no append status, crc autoflush, no addr check
133  spiWriteRegister(RH_CC110_REG_08_PKTCTRL0, RH_CC110_PKT_FORMAT_NORMAL | RH_CC110_CRC_EN | RH_CC110_LENGTH_CONFIG_VARIABLE);
134  spiWriteRegister(RH_CC110_REG_13_MDMCFG1, RH_CC110_NUM_PREAMBLE_4); // 4 preamble bytes, chan spacing not used
135  spiWriteRegister(RH_CC110_REG_17_MCSM1, RH_CC110_CCA_MODE_RSSI_PACKET | RH_CC110_RXOFF_MODE_RX | RH_CC110_TXOFF_MODE_IDLE);
136  spiWriteRegister(RH_CC110_REG_18_MCSM0, RH_CC110_FS_AUTOCAL_FROM_IDLE | RH_CC110_PO_TIMEOUT_64); // cal when going to tx or rx
137  spiWriteRegister(RH_CC110_REG_20_WORCTRL, 0xfb); // from smartrf
138  spiWriteRegister(RH_CC110_REG_29_FSTEST, 0x59); // from smartrf
139  spiWriteRegister(RH_CC110_REG_2A_PTEST, 0x7f); // from smartrf
140  spiWriteRegister(RH_CC110_REG_2B_AGCTEST, 0x3f); // from smartrf
141 
142  // Set some reasonable default values
143  uint8_t syncWords[] = { 0xd3, 0x91 };
144  setSyncWords(syncWords, sizeof(syncWords));
145  setTxPower(TransmitPower5dBm);
146  setFrequency(915.0);
147  setModemConfig(GFSK_Rb1_2Fd5_2);
148  return true;
149 }

156 // C++ level interrupt handler for this instance
157 // We use this to get RxDone and TxDone interrupts
158 void RH_CC110::handleInterrupt()
159 {
160 // Serial.println("I");
161  if (_mode == RHModeRx)
162  {
163  // Radio is configured to stay in RX until we move it to IDLE after a CRC_OK message for us
164  // We only get interrupts in RX mode, on CRC_OK
165 
166  uint8_t raw_rssi = spiBurstReadRegister(RH_CC110_REG_34_RSSI); // Was set when sync word was detected
167  // Conversion of RSSI value to received power level in dBm per TI section 5.18.2
168  if (raw_rssi >= 128)
169  _lastRssi = (((int16_t)raw_rssi - 256) / 2) - 74;
170  else
171  _lastRssi = ((int16_t)raw_rssi / 2) - 74;
172 
173  _bufLen = spiReadRegister(RH_CC110_REG_3F_FIFO);
174  if (_bufLen < 4)
175  {
176  // Something wrong there, flush the FIFO
177  spiCommand(RH_CC110_STROBE_3A_SFRX);
178  clearRxBuf();
179  return;
180  }
181  spiBurstRead(RH_CC110_REG_3F_FIFO | RH_CC110_SPI_BURST_MASK | RH_CC110_SPI_READ_MASK, _buf, _bufLen);
182  // All good so far. See if its for us
183  validateRxBuf();
184  if (_rxBufValid)
185  setModeIdle(); // Done
186  }
187 }

287 {

290

305 bool RH_CC110::send(const uint8_t* data, uint8_t len)
306 {
307  if (len > RH_CC110_MAX_MESSAGE_LEN)
308  return false;
309 
310  waitPacketSent(); // Make sure we dont interrupt an outgoing message
311  setModeIdle();
312 
313  if (!waitCAD())
314  return false; // Check channel activity
315 
316  spiWriteRegister(RH_CC110_REG_3F_FIFO, len + RH_CC110_HEADER_LEN);
317  spiWriteRegister(RH_CC110_REG_3F_FIFO,_txHeaderTo);
318  spiWriteRegister(RH_CC110_REG_3F_FIFO,_txHeaderFrom);
319  spiWriteRegister(RH_CC110_REG_3F_FIFO,_txHeaderId);
320  spiWriteRegister(RH_CC110_REG_3F_FIFO,_txHeaderFlags);
321  spiBurstWriteRegister(RH_CC110_REG_3F_FIFO, data, len);
322 
323  // Radio returns to Idle when TX is finished
324  // need waitPacketSent() to detect change of _mode and TX completion
325  setModeTx();
326 
327  return true;
328 }

412 bool RH_CC110::setFrequency(float centre)
413 {
414  // From section 5.21: fcarrier = fxosc / 2^16 * FREQ
415  uint32_t FREQ;
416  float fxosc = _is27MHz ? 27.0 : 26.0;
417  FREQ = (uint32_t)(centre * 65536 / fxosc);
418  // Some trivial checks
419  if (FREQ & 0xff000000)
420  return false;
421  spiWriteRegister(RH_CC110_REG_0D_FREQ2, (FREQ >> 16) & 0xff);
422  spiWriteRegister(RH_CC110_REG_0E_FREQ1, (FREQ >> 8) & 0xff);
423  spiWriteRegister(RH_CC110_REG_0F_FREQ0, FREQ & 0xff);
424 
425  // Radio is configured to calibrate automatically whenever it enters RX or TX mode
426  // so no need to check for PLL lock here
427  return true;
428 }
