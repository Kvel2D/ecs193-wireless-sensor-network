24 bool RH_E32::init()
25 {
26  // When a message is available, Aux will go low 5 msec before the first character is output
27  // So if we ever wait more than this period of time after Aux low, can conclude there will be no data
28  _s->setTimeout(10);

47

50 
51  return true;

73 bool RH_E32::writeParameters(Parameters& params, bool save)
74 {
75  setOperatingMode(ModeSleep);
76  params.head = save ? RH_E32_COMMAND_WRITE_PARAMS_SAVE : RH_E32_COMMAND_WRITE_PARAMS_NOSAVE;
77  // printBuffer("writing now", (uint8_t*)&params, sizeof(params));
78  size_t result = _s->write((uint8_t*)&params, sizeof(params));
79  if (result != sizeof(params))
80  return false;
81 
82  // Now we expect to get the same data back
83  result = _s->readBytes((char*)&params, sizeof(params));
84  if (result != sizeof(params))
85  return false;
86  // printBuffer("additional read", (uint8_t*)&params, sizeof(params));
87  // Without a little delay here, writing params often fails
88  delay(20);
89 
90  setOperatingMode(ModeNormal);
91  return result == sizeof(params);

94 void RH_E32::setOperatingMode(OperatingMode mode)
95 {
96  waitAuxHigh();
97  switch (mode)

103 
104  case ModeWakeUp:

107  break;
108 
109  case ModePowerSaving:
110  digitalWrite(_m0_pin, false);
111  digitalWrite(_m1_pin, true);
112  break;
113 
114  case ModeSleep:
115  digitalWrite(_m0_pin, true);
116  digitalWrite(_m1_pin, true);
117  break;
118 
119  }
120  delay(10); // Takes a little while to start its response
121  waitAuxHigh();

127  uint8_t readVersionCommand[] = { RH_E32_COMMAND_READ_VERSION, RH_E32_COMMAND_READ_VERSION, RH_E32_COMMAND_READ_VERSION };
128  _s->write(readVersionCommand, sizeof(readVersionCommand));

141  else
142  {
143  // REVISIT: do something with it?
144  }
145  }
146  else

148  // Read failed: no module? Wrong baud?
149  return false;
150  }

168 void RH_E32::validateRxBuf()
169 {
170  if (_bufLen < RH_E32_HEADER_LEN)
171  return; // Too short to be a real message
172  if (_bufLen != _buf[0])
173  return; // Do we have all the message?
174 
175  // Extract the 4 headers
176  _rxHeaderTo = _buf[1];
177  _rxHeaderFrom = _buf[2];
178  _rxHeaderId = _buf[3];
179  _rxHeaderFlags = _buf[4];
180  if (_promiscuous ||
181  _rxHeaderTo == _thisAddress ||
182  _rxHeaderTo == RH_BROADCAST_ADDRESS)
183  {
184  _rxGood++;
185  _rxBufValid = true;
186  }
187 }

195 bool RH_E32::available()
196 {
197  // Caution: long packets could be sent in several bursts
198  if (!_rxBufValid)
199  {
200  if (_mode == RHModeTx)
201  return false;
202 
203  if (!_s->available())
204  return false;
205 
206  // Suck up all the characters we can
207  uint8_t data;
208  while (_s->readBytes((char *)&data, 1) == 1) // Not read timeout
209  {
210  _buf[_bufLen++] = data;
211  }
212  // Now assess what we have
213  if (_bufLen < RH_E32_HEADER_LEN)
214  {
215  // Serial.println("Incomplete header");
216  return false;
217  }
218  else if (_bufLen < _buf[0])
219  {
220  // Serial.println("Incomplete message");
221  return false;
222  }
223  else if ( _bufLen > _buf[0]
224  || _bufLen > RH_E32_MAX_PAYLOAD_LEN)
225  {
226  // Serial.println("Overrun");
227  clearRxBuf();
228  _rxBad++;
229  return false;
230  }
231 
232  // Else it a partial or complete message, test it
233  // printBuffer("read success", _buf, _bufLen);
234  validateRxBuf();
235  }
236  return _rxBufValid;
237 }

239 bool RH_E32::recv(uint8_t* buf, uint8_t* len)
240 {
241  if (!available())
242  return false;
243  if (buf && len)
244  {
245  // Skip the 4 headers that are at the beginning of the rxBuf
246  if (*len > _bufLen - RH_E32_HEADER_LEN)
247  *len = _bufLen - RH_E32_HEADER_LEN;
248  memcpy(buf, _buf + RH_E32_HEADER_LEN, *len);
249  }
250  clearRxBuf(); // This message accepted and cleared
251  return true;
252 }

254 bool RH_E32::send(const uint8_t* data, uint8_t len)
255 {
256  if (len > RH_E32_MAX_MESSAGE_LEN)
257  return false;
258 
259  waitPacketSent(); // Make sure we dont collide with previous message
260 
261  // Set up the headers
262  _buf[0] = len + RH_E32_HEADER_LEN; // Number of octets in teh whole message
263  _buf[1] = _txHeaderTo;
264  _buf[2] = _txHeaderFrom;
265  _buf[3] = _txHeaderId;
266  _buf[4] = _txHeaderFlags;
267 
268  // REVISIT: do we really have to do this? perhaps just write it after writing the header?
269  memcpy(_buf+RH_E32_HEADER_LEN, data, len);
270 
271  _s->write(_buf, len + RH_E32_HEADER_LEN);
272  setMode(RHModeTx);
273  _txGood++;
274  // Aux will return high when the TX buffer is empty
275 
276  return true;
277 }
