22 bool RH_NRF51::init()
23 {
24  // Enable the High Frequency clock to the system as a whole
25  NRF_CLOCK->EVENTS_HFCLKSTARTED = 0;
26  NRF_CLOCK->TASKS_HFCLKSTART = 1;
27  /* Wait for the external oscillator to start up */
28  while (NRF_CLOCK->EVENTS_HFCLKSTARTED == 0)
29  ;
30 
31  // Disable and reset the radio
32  NRF_RADIO->POWER = RADIO_POWER_POWER_Disabled;
33  NRF_RADIO->POWER = RADIO_POWER_POWER_Enabled;
34  NRF_RADIO->EVENTS_DISABLED = 0;
35  NRF_RADIO->TASKS_DISABLE = 1;
36  // Wait until we are in DISABLE state
37  while (NRF_RADIO->EVENTS_DISABLED == 0) {}
38 
39  // Physical on-air address is set in PREFIX0 + BASE0 by setNetworkAddress
40  NRF_RADIO->TXADDRESS = 0x00; // Use logical address 0 (PREFIX0 + BASE0)
41  NRF_RADIO->RXADDRESSES = 0x01; // Enable reception on logical address 0 (PREFIX0 + BASE0)
42 
43  // Configure the CRC
44  NRF_RADIO->CRCCNF = (RADIO_CRCCNF_LEN_Two << RADIO_CRCCNF_LEN_Pos); // Number of checksum bits
45  NRF_RADIO->CRCINIT = 0xFFFFUL; // Initial value
46  NRF_RADIO->CRCPOLY = 0x11021UL; // CRC poly: x^16+x^12^x^5+1
47 
48  // These shorts will make the radio transition from Ready to Start to Disable automatically
49  // for both TX and RX, which makes for much shorter on-air times
50  NRF_RADIO->SHORTS = (RADIO_SHORTS_READY_START_Enabled << RADIO_SHORTS_READY_START_Pos)
51  | (RADIO_SHORTS_END_DISABLE_Enabled << RADIO_SHORTS_END_DISABLE_Pos);
52 
53  NRF_RADIO->PCNF0 = (8 << RADIO_PCNF0_LFLEN_Pos) // Payload size length in bits
54  | (1 << RADIO_PCNF0_S0LEN_Pos) // S0 is 1 octet
55  | (8 << RADIO_PCNF0_S1LEN_Pos); // S1 is 1 octet
56 
57  // Make sure we are powered down
58  setModeIdle();
59 
60  // Set a default network address
61  uint8_t default_network_address[] = {0xE7, 0xE7, 0xE7, 0xE7, 0xE7};
62  setNetworkAddress(default_network_address, sizeof(default_network_address));
63 
64  setChannel(2); // The default, in case it was set by another app without powering down
65  setRF(RH_NRF51::DataRate2Mbps, RH_NRF51::TransmitPower0dBm);
66  setEncryptionKey(NULL);
67  return true;
68 }

76 bool RH_NRF51::setNetworkAddress(uint8_t* address, uint8_t len)
77 {
78  if (len < 3 || len > 5)
79  return false;
80 
81  // First byte is the prefix, remainder are base
82  NRF_RADIO->PREFIX0 = ((address[0] << RADIO_PREFIX0_AP0_Pos) & RADIO_PREFIX0_AP0_Msk);
83  uint32_t base;
84  memcpy(&base, address+1, len-1);
85  NRF_RADIO->BASE0 = base;
86 
87  NRF_RADIO->PCNF1 = (
88  (((sizeof(_buf)) << RADIO_PCNF1_MAXLEN_Pos) & RADIO_PCNF1_MAXLEN_Msk) // maximum length of payload
89  | (((0UL) << RADIO_PCNF1_STATLEN_Pos) & RADIO_PCNF1_STATLEN_Msk) // expand the payload with 0 bytes
90  | (((len-1) << RADIO_PCNF1_BALEN_Pos) & RADIO_PCNF1_BALEN_Msk)); // base address length in number of bytes.
91 
92  return true;
93 }

95 bool RH_NRF51::setRF(DataRate data_rate, TransmitPower power)
96 {
97  uint8_t mode;
98  uint8_t p;
99 
100  if (data_rate == DataRate2Mbps)
101  mode = RADIO_MODE_MODE_Nrf_2Mbit;
102  else if (data_rate == DataRate1Mbps)
103  mode = RADIO_MODE_MODE_Nrf_1Mbit;
104  else if (data_rate == DataRate250kbps)
105  mode = RADIO_MODE_MODE_Nrf_250Kbit;
106  else
107  return false;// Invalid
108 
109  if (power == TransmitPower4dBm)
110  p = RADIO_TXPOWER_TXPOWER_Pos4dBm;
111  else if (power == TransmitPower0dBm)
112  p = RADIO_TXPOWER_TXPOWER_0dBm;
113  else if (power == TransmitPowerm4dBm)
114  p = RADIO_TXPOWER_TXPOWER_Neg4dBm;
115  else if (power == TransmitPowerm8dBm)
116  p = RADIO_TXPOWER_TXPOWER_Neg8dBm;
117  else if (power == TransmitPowerm12dBm)
118  p = RADIO_TXPOWER_TXPOWER_Neg12dBm;
119  else if (power == TransmitPowerm16dBm)
120  p = RADIO_TXPOWER_TXPOWER_Neg16dBm;
121  else if (power == TransmitPowerm20dBm)
122  p = RADIO_TXPOWER_TXPOWER_Neg20dBm;
123  else if (power == TransmitPowerm30dBm)
124  p = RADIO_TXPOWER_TXPOWER_Neg30dBm;
125  else
126  return false; // Invalid
127 
128 
129  NRF_RADIO->TXPOWER = ((p << RADIO_TXPOWER_TXPOWER_Pos) & RADIO_TXPOWER_TXPOWER_Msk);
130  NRF_RADIO->MODE = ((mode << RADIO_MODE_MODE_Pos) & RADIO_MODE_MODE_Msk);
131 
132  return true;

148 void RH_NRF51::setModeRx()
149 {
150  if (_mode != RHModeRx)
151  {
152  setModeIdle(); // Can only start RX from DISABLE state
153 
154 #if RH_NRF51_HAVE_ENCRYPTION
155  // Maybe set the AES CCA module for the correct encryption mode
156  if (_encrypting)
157  NRF_CCM->MODE = (CCM_MODE_MODE_Decryption << CCM_MODE_MODE_Pos); // Decrypt
158  NRF_CCM->MICSTATUS = 0;
159 #endif
160 
161  // Radio will transition automatically to Disable state when a message is received
162  NRF_RADIO->PACKETPTR = (uint32_t)_buf;
163  NRF_RADIO->EVENTS_READY = 0U;
164  NRF_RADIO->TASKS_RXEN = 1;
165  NRF_RADIO->EVENTS_END = 0U; // So we can detect end of reception
166  _mode = RHModeRx;

170 void RH_NRF51::setModeTx()
171 {
172  if (_mode != RHModeTx)
173  {
174  setModeIdle(); // Can only start RX from DISABLE state
175 
176  // Sigh: it seems that it takes longer to start the receiver than the transmitter for this type
177  // of radio, so if a message is received and an ACK or reply is sent to soon, the original transmitter
178  // may not see the reply. So we delay here to make sure the receiver is ready.
179  // Yes, I know this is very ugly
180  delay(1);
181 
182 #if RH_NRF51_HAVE_ENCRYPTION
183  // Maybe set the AES CCA module for the correct encryption mode
184  if (_encrypting)
185  NRF_CCM->MODE = (CCM_MODE_MODE_Encryption << CCM_MODE_MODE_Pos); // Encrypt
186 #endif
187  // Radio will transition automatically to Disable state at the end of transmission
188  NRF_RADIO->PACKETPTR = (uint32_t)_buf;
189  NRF_RADIO->EVENTS_READY = 0U;
190  NRF_RADIO->TASKS_TXEN = 1;
191  NRF_RADIO->EVENTS_END = 0U; // So we can detect end of transmission
192  _mode = RHModeTx;
193  }

196 bool RH_NRF51::send(const uint8_t* data, uint8_t len)
197 {
198  if (len > RH_NRF51_MAX_MESSAGE_LEN)
199  return false;
200 
201 #if RH_NRF51_HAVE_ENCRYPTION
202  if (_encrypting && len > RH_NRF51_MAX_ENCRYPTED_MESSAGE_LEN)
203  return false;
204 #endif
205 
206  if (!waitCAD())
207  return false; // Check channel activity
208 
209  // Set up the headers
210  _buf[0] = 0; // S0
211  _buf[1] = len + RH_NRF51_HEADER_LEN;
212  _buf[2] = 0; // S1
213  // The following octets are subject to encryption
214  _buf[3] = _txHeaderTo;
215  _buf[4] = _txHeaderFrom;
216  _buf[5] = _txHeaderId;
217  _buf[6] = _txHeaderFlags;
218  memcpy(_buf+RH_NRF51_HEADER_LEN, data, len);
219  _rxBufValid = false;
220  setModeTx();
221 
222  // Radio will return to Disabled state after transmission is complete
223  _txGood++;
224  return true;
225 }

227 bool RH_NRF51::waitPacketSent()
228 {
229  // If we are not currently in transmit mode, there is no packet to wait for
230  if (_mode != RHModeTx)
231  return false;
232 
233  // When the Disabled event occurs we know the transmission has completed
234  while (!NRF_RADIO->EVENTS_END)
235  {
236  YIELD;
237  }
238  setModeIdle();
239 
240  return true;

265 // Check whether the latest received message is complete and uncorrupted
266 void RH_NRF51::validateRxBuf()
267 {
268  if (_buf[1] < RH_NRF51_HEADER_LEN)
269  return; // Too short to be a real message
270  // Extract the 4 headers following S0, LEN and S1
271  _rxHeaderTo = _buf[3];
272  _rxHeaderFrom = _buf[4];
273  _rxHeaderId = _buf[5];
274  _rxHeaderFlags = _buf[6];
275 
276  if (_promiscuous ||
277  _rxHeaderTo == _thisAddress ||
278  _rxHeaderTo == RH_BROADCAST_ADDRESS)
279  {
280  _rxGood++;
281  _rxBufValid = true;
282  }

285 void RH_NRF51::setEncryptionKey(uint8_t* key)
286 {
287 #if RH_NRF51_HAVE_ENCRYPTION
288  if (key)
289  {
290  // Configure for on-the-fly encryption
291  // Set the key
292  memset(_encryption_cnf, 0, sizeof(_encryption_cnf));
293  memcpy(_encryption_cnf, key, RH_NRF51_ENCRYPTION_KEY_LENGTH);
294  // AES configuration data area
295  // Note that the IV (Nonce) is not set, defaults to 0s
296  NRF_CCM->CNFPTR = (uint32_t)_encryption_cnf;
297 
298  // Set AES CCM input and putput buffers
299  // Make sure the _buf is encrypted and put back into _buf
300  NRF_CCM->INPTR = (uint32_t)_buf;
301  NRF_CCM->OUTPTR = (uint32_t)_buf;
302  // Also need to set SCRATCHPTR temp buffer os size 16+MAXPACKETSIZE in RAM
303  // FIXME: shared buffers if several radios
304  NRF_CCM->SCRATCHPTR = (uint32_t)_scratch;
305 
306  // SHORT from RADIO READY to AESCCM KSGEN using PPI predefined channel 24
307  // Also RADIO ADDRESS to AESCCM CRYPT using PPI predefined channel 25
308  NRF_PPI->CHENSET = (PPI_CHENSET_CH24_Enabled << PPI_CHENSET_CH24_Pos)
309  | (PPI_CHENSET_CH25_Enabled << PPI_CHENSET_CH25_Pos)
310  ;
311 
312  // SHORT from AESCCM ENDKSGEN to AESCCM CRYPT
313  NRF_CCM->SHORTS = (CCM_SHORTS_ENDKSGEN_CRYPT_Enabled << CCM_SHORTS_ENDKSGEN_CRYPT_Pos);
314 
315  // Enable the CCM module
316  NRF_CCM->ENABLE = (CCM_ENABLE_ENABLE_Enabled << CCM_ENABLE_ENABLE_Pos);
317 
318  _encrypting = true;
319  }
320  else
321  {
322  // Disable the CCM module
323  NRF_CCM->ENABLE = (CCM_ENABLE_ENABLE_Disabled << CCM_ENABLE_ENABLE_Pos);
324  _encrypting = false;
325  }
326 #endif
327 }

329 bool RH_NRF51::available()
330 {
331  if (!_rxBufValid)
332  {
333  if (_mode == RHModeTx)
334  return false;
335  setModeRx();
336  if (!NRF_RADIO->EVENTS_END)
337  return false; // No message yet
338  setModeIdle();
339 #if RH_NRF51_HAVE_ENCRYPTION
340  // If encryption is enabled, the decrypted message is not available yet, and there seems
341  // to be no way to be sure when its ready, but a delay of 2ms is enough
342  if (_encrypting)
343  delay(2);
344 #endif
345  if (!NRF_RADIO->CRCSTATUS)
346  {
347  // Bad CRC, restart the radio
348  _rxBad++;
349  setModeRx();
350  return false;
351  }
352  validateRxBuf();
353  if (!_rxBufValid)
354  setModeRx(); // Try for another
355  }
356  return _rxBufValid;
357 }

365 bool RH_NRF51::recv(uint8_t* buf, uint8_t* len)
366 {
367  if (!available())
368  return false;
369  if (buf && len)
370  {
371  // Skip the 4 headers that are at the beginning of the rxBuf
372  // the payload length is the first octet in _buf
373  if (*len > _buf[1]-RH_NRF51_HEADER_LEN)
374  *len = _buf[1]-RH_NRF51_HEADER_LEN;
375  memcpy(buf, _buf+RH_NRF51_HEADER_LEN, *len);
376  }
377  clearRxBuf(); // This message accepted and cleared
378  return true;
379 }
