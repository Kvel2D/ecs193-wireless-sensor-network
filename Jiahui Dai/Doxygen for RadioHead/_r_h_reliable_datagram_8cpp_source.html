48 bool RHReliableDatagram::sendtoWait(uint8_t* buf, uint8_t len, uint8_t address)
49 {
50  // Assemble the message
51  uint8_t thisSequenceNumber = ++_lastSequenceNumber;
52  uint8_t retries = 0;
53  while (retries++ <= _retries)
54  {
55  setHeaderId(thisSequenceNumber);
56  setHeaderFlags(RH_FLAGS_NONE, RH_FLAGS_ACK); // Clear the ACK flag
57  sendto(buf, len, address);
58  waitPacketSent();
59 
60  // Never wait for ACKS to broadcasts:
61  if (address == RH_BROADCAST_ADDRESS)
62  return true;
63 
64  if (retries > 1)
65  _retransmissions++;
66  unsigned long thisSendTime = millis(); // Timeout does not include original transmit time
67 
68  // Compute a new timeout, random between _timeout and _timeout*2
69  // This is to prevent collisions on every retransmit
70  // if 2 nodes try to transmit at the same time
71 #if (RH_PLATFORM == RH_PLATFORM_RASPI) // use standard library random(), bugs in random(min, max)
72  uint16_t timeout = _timeout + (_timeout * (random() & 0xFF) / 256);
73 #else
74  uint16_t timeout = _timeout + (_timeout * random(0, 256) / 256);
75 #endif
76  int32_t timeLeft;
77  while ((timeLeft = timeout - (millis() - thisSendTime)) > 0)
78  {
79  if (waitAvailableTimeout(timeLeft))
80  {
81  uint8_t from, to, id, flags;
82  if (recvfrom(0, 0, &from, &to, &id, &flags)) // Discards the message
83  {
84  // Now have a message: is it our ACK?
85  if ( from == address
86  && to == _thisAddress
87  && (flags & RH_FLAGS_ACK)
88  && (id == thisSequenceNumber))
89  {
90  // Its the ACK we are waiting for
91  return true;
92  }
93  else if ( !(flags & RH_FLAGS_ACK)
94  && (id == _seenIds[from]))
95  {
96  // This is a request we have already received. ACK it again
97  acknowledge(id, from);
98  }
99  // Else discard it
100  }
101  }
102  // Not the one we are waiting for, maybe keep waiting until timeout exhausted
103  YIELD;
104  }
105  // Timeout exhausted, maybe retry
106  YIELD;
107  }
108  // Retries exhausted
109  return false;
110 }

113 bool RHReliableDatagram::recvfromAck(uint8_t* buf, uint8_t* len, uint8_t* from, uint8_t* to, uint8_t* id, uint8_t* flags)
114 {
115  uint8_t _from;
116  uint8_t _to;
117  uint8_t _id;
118  uint8_t _flags;
119  // Get the message before its clobbered by the ACK (shared rx and tx buffer in some drivers
120  if (available() && recvfrom(buf, len, &_from, &_to, &_id, &_flags))
121  {
122  // Never ACK an ACK
123  if (!(_flags & RH_FLAGS_ACK))
124  {
125  // Its a normal message not an ACK
126  if (_to ==_thisAddress)
127  {
128  // Its for this node and
129  // Its not a broadcast, so ACK it
130  // Acknowledge message with ACK set in flags and ID set to received ID
131  acknowledge(_id, _from);
132  }
133  // If we have not seen this message before, then we are interested in it
134  if (_id != _seenIds[_from])
135  {
136  if (from) *from = _from;
137  if (to) *to = _to;
138  if (id) *id = _id;
139  if (flags) *flags = _flags;
140  _seenIds[_from] = _id;
141  return true;
142  }
143  // Else just re-ack it and wait for a new one
144  }
145  }
146  // No message for us available
147  return false;
148 }

176 void RHReliableDatagram::acknowledge(uint8_t id, uint8_t from)
177 {
178  setHeaderId(id);
179  setHeaderFlags(RH_FLAGS_ACK);
180  // We would prefer to send a zero length ACK,
181  // but if an RH_RF22 receives a 0 length message with a CRC error, it will never receive
182  // a 0 length message again, until its reset, which makes everything hang :-(
183  // So we send an ACK of 1 octet
184  // REVISIT: should we send the RSSI for the information of the sender?
185  uint8_t ack = '!';
186  sendto(&ack, sizeof(ack), from);
187  waitPacketSent();
188 }
