58 uint8_t RHNRFSPIDriver::spiWrite(uint8_t reg, uint8_t val)
59 {
60  uint8_t status = 0;
61  ATOMIC_BLOCK_START;
62  _spi.beginTransaction();
63  digitalWrite(_slaveSelectPin, LOW);
64  status = _spi.transfer(reg); // Send the address
65  _spi.transfer(val); // New value follows
66 #if (RH_PLATFORM == RH_PLATFORM_ARDUINO) && defined(__arm__) && defined(CORE_TEENSY)
67  // Sigh: some devices, such as MRF89XA dont work properly on Teensy 3.1:
68  // At 1MHz, the clock returns low _after_ slave select goes high, which prevents SPI
69  // write working. This delay gixes time for the clock to return low.
70 delayMicroseconds(5);
71 #endif
72  digitalWrite(_slaveSelectPin, HIGH);
73  _spi.endTransaction();
74  ATOMIC_BLOCK_END;
75  return status;
76 }
