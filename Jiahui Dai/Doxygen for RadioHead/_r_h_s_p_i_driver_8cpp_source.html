30 uint8_t RHSPIDriver::spiRead(uint8_t reg)
31 {
32  uint8_t val;
33  ATOMIC_BLOCK_START;
34  digitalWrite(_slaveSelectPin, LOW);
35  _spi.transfer(reg & ~RH_SPI_WRITE_MASK); // Send the address with the write mask off
