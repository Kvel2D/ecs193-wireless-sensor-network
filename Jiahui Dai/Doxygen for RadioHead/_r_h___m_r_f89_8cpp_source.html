48 bool RH_MRF89::init()
49 {
50  // MRF89 data cant handle SPI greater than 1MHz.
51  // Sigh on teensy at 1MHz, need special delay after writes, see RHNRFSPIDriver::spiWrite
52  _spi.setFrequency(RHGenericSPI::Frequency1MHz);
53  if (!RHNRFSPIDriver::init())
54  return false;
55 
56  // Initialise the chip select pins
57  pinMode(_csconPin, OUTPUT);
58  digitalWrite(_csconPin, HIGH);
59  pinMode(_csdatPin, OUTPUT);
60  digitalWrite(_csdatPin, HIGH);
61 
62  // Determine the interrupt number that corresponds to the interruptPin
63  int interruptNumber = digitalPinToInterrupt(_interruptPin);
64  if (interruptNumber == NOT_AN_INTERRUPT)
65  return false;
66 #ifdef RH_ATTACHINTERRUPT_TAKES_PIN_NUMBER
67  interruptNumber = _interruptPin;
68 #endif
69 
70  // Tell the low level SPI interface we will use SPI within this interrupt
71  spiUsingInterrupt(interruptNumber);
72 
73  // Make sure we are not in some unexpected mode from a previous run
74  setOpMode(RH_MRF89_CMOD_STANDBY);
75 
76  // No way to check the device type but lets trivially check there is something there
77  // by trying to change a register:
78  spiWriteRegister(RH_MRF89_REG_02_FDEVREG, 0xaa);
79  if (spiReadRegister(RH_MRF89_REG_02_FDEVREG) != 0xaa)
80  return false;
81  spiWriteRegister(RH_MRF89_REG_02_FDEVREG, 0x3); // Back to the default for FDEV
82  if (spiReadRegister(RH_MRF89_REG_02_FDEVREG) != 0x3)
83  return false;
84 
85  // Add by Adrien van den Bossche <vandenbo@univ-tlse2.fr> for Teensy
86  // ARM M4 requires the below. else pin interrupt doesn't work properly.
87  // On all other platforms, its innocuous, belt and braces
88  pinMode(_interruptPin, INPUT);
89 
90  // Set up interrupt handler
91  // Since there are a limited number of interrupt glue functions isr*() available,
92  // we can only support a limited number of devices simultaneously
93  // On some devices, notably most Arduinos, the interrupt pin passed in is actually the
94  // interrupt number. You have to figure out the interruptnumber-to-interruptpin mapping
95  // yourself based on knowledge of what Arduino board you are running on.
96  if (_myInterruptIndex == 0xff)
97  {
98  // First run, no interrupt allocated yet
99  if (_interruptCount <= RH_MRF89_NUM_INTERRUPTS)
100  _myInterruptIndex = _interruptCount++;
101  else
102  return false; // Too many devices, not enough interrupt vectors
103  }
104  _deviceForInterrupt[_myInterruptIndex] = this;
105  if (_myInterruptIndex == 0)
106  attachInterrupt(interruptNumber, isr0, RISING);
107  else if (_myInterruptIndex == 1)
108  attachInterrupt(interruptNumber, isr1, RISING);
109  else if (_myInterruptIndex == 2)
110  attachInterrupt(interruptNumber, isr2, RISING);
111  else
112  return false; // Too many devices, not enough interrupt vectors
113 
114  // When used with the MRF89XAM9A module, per 75017B.pdf section 1.3, need:
115  // crystal freq = 12.8MHz
116  // clock output disabled
117  // frequency bands 902-915 or 915-928
118  // VCOT 60mV
119  // OOK max 28kbps
120  // Based on 70622C.pdf, section 3.12:
121  spiWriteRegister(RH_MRF89_REG_00_GCONREG, RH_MRF89_CMOD_STANDBY | RH_MRF89_FBS_950_960 | RH_MRF89_VCOT_60MV);
122  spiWriteRegister(RH_MRF89_REG_01_DMODREG, RH_MRF89_MODSEL_FSK | RH_MRF89_OPMODE_PACKET); // FSK, Packet mode, LNA 0dB
123  spiWriteRegister(RH_MRF89_REG_02_FDEVREG, 0); // Set by setModemConfig
124  spiWriteRegister(RH_MRF89_REG_03_BRSREG, 0); // Set by setModemConfig
125  spiWriteRegister(RH_MRF89_REG_04_FLTHREG, 0); // Set by setModemConfig (OOK only)
126  spiWriteRegister(RH_MRF89_REG_05_FIFOCREG, RH_MRF89_FSIZE_64);
127  spiWriteRegister(RH_MRF89_REG_06_R1CREG, 0); // Set by setFrequency
128  spiWriteRegister(RH_MRF89_REG_07_P1CREG, 0); // Set by setFrequency
129  spiWriteRegister(RH_MRF89_REG_08_S1CREG, 0); // Set by setFrequency
130  spiWriteRegister(RH_MRF89_REG_09_R2CREG, 0); // Frequency set 2 not used
131  spiWriteRegister(RH_MRF89_REG_0A_P2CREG, 0); // Frequency set 2 not used
132  spiWriteRegister(RH_MRF89_REG_0B_S2CREG, 0); // Frequency set 2 not used
133  spiWriteRegister(RH_MRF89_REG_0C_PACREG, RH_MRF89_PARC_23);
134  // IRQ0 rx mode: SYNC (not used)
135  // IRQ1 rx mode: CRCOK
136  // IRQ1 tx mode: TXDONE
137  spiWriteRegister(RH_MRF89_REG_0D_FTXRXIREG, RH_MRF89_IRQ0RXS_PACKET_SYNC | RH_MRF89_IRQ1RXS_PACKET_CRCOK | RH_MRF89_IRQ1TX);
138  spiWriteRegister(RH_MRF89_REG_0E_FTPRIREG, RH_MRF89_LENPLL);
139  spiWriteRegister(RH_MRF89_REG_0F_RSTHIREG, 0x00); // default not used if no RSSI interrupts
140  spiWriteRegister(RH_MRF89_REG_10_FILCREG, 0); // Set by setModemConfig
141 
142  spiWriteRegister(RH_MRF89_REG_11_PFCREG, 0x38);// 100kHz, recommended, but not used, see RH_MRF89_REG_12_SYNCREG OOK only?
143  spiWriteRegister(RH_MRF89_REG_12_SYNCREG, RH_MRF89_SYNCREN | RH_MRF89_SYNCWSZ_32); // No polyphase, no bsync, sync, 0 errors
144  spiWriteRegister(RH_MRF89_REG_13_RSVREG, 0x07);//default
145 // spiWriteRegister(RH_MRF89_REG_14_RSTSREG, 0x00); // NO, read only
146  spiWriteRegister(RH_MRF89_REG_15_OOKCREG, 0x00); // Set by setModemConfig OOK only
147  spiWriteRegister(RH_MRF89_REG_16_SYNCV31REG, 0x69); // Set by setSyncWords
148  spiWriteRegister(RH_MRF89_REG_17_SYNCV23REG, 0x81); // Set by setSyncWords
149  spiWriteRegister(RH_MRF89_REG_18_SYNCV15REG, 0x7E); // Set by setSyncWords
150  spiWriteRegister(RH_MRF89_REG_19_SYNCV07REG, 0x96); // Set by setSyncWords
151  // TXIPOLFV set by setModemConfig. power set by setTxPower
152  spiWriteRegister(RH_MRF89_REG_1A_TXCONREG, 0xf0 | RH_MRF89_TXOPVAL_13DBM); // TX cutoff freq=375kHz,
153  spiWriteRegister(RH_MRF89_REG_1B_CLKOREG, 0x00); // Disable clock output to save power
154  spiWriteRegister(RH_MRF89_REG_1C_PLOADREG, 0x40); // payload=64bytes (no RX-filtering on packet length)
155  spiWriteRegister(RH_MRF89_REG_1D_NADDSREG, 0x00); // Node Address (0=default) Not used
156  spiWriteRegister(RH_MRF89_REG_1E_PKTCREG, RH_MRF89_PKTLENF | RH_MRF89_PRESIZE_4 | RH_MRF89_WHITEON | RH_MRF89_CHKCRCEN | RH_MRF89_ADDFIL_OFF);
157  spiWriteRegister(RH_MRF89_REG_1F_FCRCREG, 0x00); // default (FIFO access in standby=write, clear FIFO on CRC mismatch)
158 
159  // Looking OK now
160  // Set some suitable defaults:
161  setPreambleLength(3); // The default
162  uint8_t syncwords[] = { 0x69, 0x81, 0x7e, 0x96 }; // Same as RH_MRF89XA
163  setSyncWords(syncwords, sizeof(syncwords));
164  setTxPower(RH_MRF89_TXOPVAL_1DBM);
165  if (!setFrequency(915.4))
166  return false;
167  // Some slow, reliable default speed and modulation
168  if (!setModemConfig(FSK_Rb20Fd40))
169  return false;
170 
171  return true;

188 // C++ level interrupt handler for this instance
189 // MRF89XA is unusual in that it has 2 interrupt lines, and not a single, combined one.
190 // Only one of the several interrupt lines (IRQ1) from the RFM95 needs to be
191 // connnected to the processor.
192 // We use this to get CRCOK and TXDONE interrupts
193 void RH_MRF89::handleInterrupt()
194 {
195 // Serial.println("I");
196  if (_mode == RHModeTx)
197  {
198 // Serial.println("T");
199  // TXDONE
200  // Transmit is complete
201  _txGood++;
202  setModeIdle();
203  }
204  else if (_mode == RHModeRx)
205  {
206 // Serial.println("R");
207  // CRCOK
208  // We have received a packet.
209  // First byte in FIFO is packet length
210 
211  // REVISIT: Capture last rssi from RSTSREG
212  // based roughly on Figure 3-9
213  _lastRssi = (spiReadRegister(RH_MRF89_REG_14_RSTSREG) >> 1) - 120;
214 
215  _bufLen = spiReadData();
216  if (_bufLen < 4)
217  {
218  // Drain the FIFO
219  uint8_t i;
220  for (i = 0; spiReadRegister(RH_MRF89_REG_0D_FTXRXIREG) & RH_MRF89_FIFOEMPTY; i++)
221  spiReadData();
222  clearRxBuf();
223  return;
224  }
225 
226  // Now drain all the data from the FIFO into _buf
227  uint8_t i;
228  for (i = 0; spiReadRegister(RH_MRF89_REG_0D_FTXRXIREG) & RH_MRF89_FIFOEMPTY; i++)
229  _buf[i] = spiReadData();
230 
231  // All good. See if its for us
232  validateRxBuf();
233  if (_rxBufValid)
234  setModeIdle(); // Got one
235  }
236 }

285 {

373 {

391 bool RH_MRF89::send(const uint8_t* data, uint8_t len)
392 {
393  if (len > RH_MRF89_MAX_MESSAGE_LEN)
394  return false;
395 
396  waitPacketSent(); // Make sure we dont interrupt an outgoing message
397  setModeIdle();
398 
399  if (!waitCAD())
400  return false; // Check channel activity
401 
402  // First octet is the length of the chip payload
403  // 0 length messages are transmitted but never trigger a receive!
404  spiWriteData(len + RH_MRF89_HEADER_LEN);
405  spiWriteData(_txHeaderTo);
406  spiWriteData(_txHeaderFrom);
407  spiWriteData(_txHeaderId);
408  spiWriteData(_txHeaderFlags);
409  spiWriteData(data, len);
410  setModeTx(); // Start transmitting
411 
412  return true;
413 }

447 bool RH_MRF89::verifyPLLLock()
448 {
449  // Verify PLL-lock per instructions in Note 1 section 3.12
450  // Need to do this after changing frequency.
451  uint8_t ftpriVal = spiReadRegister(RH_MRF89_REG_0E_FTPRIREG);
452  spiWriteRegister(RH_MRF89_REG_0E_FTPRIREG, ftpriVal | RH_MRF89_LSTSPLL); // Clear PLL lock bit
453  setOpMode(RH_MRF89_CMOD_FS);
454  unsigned long ulStartTime = millis();
455  while ((millis() - ulStartTime < 1000))
456  {
457  ftpriVal = spiReadRegister(RH_MRF89_REG_0E_FTPRIREG);
458  if ((ftpriVal & RH_MRF89_LSTSPLL) != 0)
459  break;
460  }
461  setOpMode(RH_MRF89_CMOD_STANDBY);
462  return ((ftpriVal & RH_MRF89_LSTSPLL) != 0);
463 }

465 bool RH_MRF89::setFrequency(float centre)
466 {
467  // REVISIT: FSK only: its different for OOK :-(
468 
469  uint8_t FBS;
470  if (centre >= 902.0 && centre < 915.0)
471  {
472  FBS = RH_MRF89_FBS_902_915;
473  }
474  else if (centre >= 915.0 && centre <= 928.0)
475  {
476  FBS = RH_MRF89_FBS_915_928;
477  }
478  else if (centre >= 950.0 && centre <= 960.0)
479  {
480  // Not all modules support this frequency band:
481  // The MRF98XAM9A does not
482  FBS = RH_MRF89_FBS_950_960;
483  }
484 // else if (centre >= 863.0 && centre <= 870.0)
485 // {
486 // // Not all modules support this frequency band:
487 // // The MRF98XAM9A does not
488 // FBS = RH_MRF89_FBS_950_960; // Yes same as above
489 // }
490  else
491  {
492  // Cant do this freq
493  return false;
494  }
495 
496  // Based on frequency calcs done in MRF89XA.h
497 // uint8_t R = 100; // Recommended
498  uint8_t R = 119; // Also recommended :-(
499  uint32_t centre_kHz = centre * 1000;
500  uint32_t xtal_kHz = (RH_MRF89_XTAL_FREQ * 1000);
501  uint32_t compare = (centre_kHz * 8 * (R + 1)) / (9 * xtal_kHz);
502  uint8_t P = ((compare - 75) / 76) + 1;
503  uint8_t S = compare - (75 * (P + 1));
504 
505  // Now set the new register values:
506  uint8_t val = spiReadRegister(RH_MRF89_REG_00_GCONREG);
507  val = (val & ~RH_MRF89_FBS) | (FBS & RH_MRF89_FBS);
508  spiWriteRegister(RH_MRF89_REG_00_GCONREG, val);
509 
510  spiWriteRegister(RH_MRF89_REG_06_R1CREG, R);
511  spiWriteRegister(RH_MRF89_REG_07_P1CREG, P);
512  spiWriteRegister(RH_MRF89_REG_08_S1CREG, S);
513 
514  return verifyPLLLock();
515 }

540

542 }

544 void RH_MRF89::setPreambleLength(uint8_t bytes)
545 {

547  {
548  bytes--;
549  uint8_t pktcreg = spiReadRegister(RH_MRF89_REG_1E_PKTCREG);
550  pktcreg = (pktcreg & ~RH_MRF89_PRESIZE) | ((bytes << 5) & RH_MRF89_PRESIZE);

552  }

555 void RH_MRF89::setSyncWords(const uint8_t* syncWords, uint8_t len)
556 {
557  if (syncWords && (len > 0 and len <= 4))
558  {
559  uint8_t syncreg = spiReadRegister(RH_MRF89_REG_12_SYNCREG);
560  syncreg = (syncreg & ~RH_MRF89_SYNCWSZ) | (((len - 1) << 3) & RH_MRF89_SYNCWSZ);
561  spiWriteRegister(RH_MRF89_REG_12_SYNCREG, syncreg);
562  uint8_t i;
563  for (i = 0; i < 4; i++)
564  {
565  if (len > i)
566  spiWriteRegister(RH_MRF89_REG_16_SYNCV31REG + i, syncWords[i]);
567  }
568  }
